-- Helper: can the current user manage team settings (owner OR coach with the flag)?
create or replace function public.can_manage_team(team_uuid uuid)
returns boolean language sql security definer stable
set search_path = public as $$
  select
    exists(select 1 from teams where id = team_uuid and created_by = auth.uid())
    OR
    exists(
      select 1 from team_members
      where team_id = team_uuid
        and user_id = auth.uid()
        and status = 'active'
        and role in ('coach', 'assistant_coach')
        and can_manage_team_settings = true
    );
$$;

-- Updated policies
create policy "managers update team" on teams for update
  using (can_manage_team(id));

-- Only the original owner can delete the team or transfer ownership
create policy "owner deletes team" on teams for delete
  using (created_by = auth.uid());

-- Managers (or owner) can manage coaches; assistant coaches and regular coaches without toggle cannot
create policy "managers manage coach rows" on team_members for all
  using (
    can_manage_team(team_id)
    and role in ('coach', 'assistant_coach')
  );
The distinction:

Owner (teams.created_by) — can do absolutely everything including transfer ownership and delete team
Manager (can_manage_team_settings = true) — can do everything the owner can EXCEPT delete team or transfer ownership
Coach without toggle — manage day-to-day (events, players, tasks, messages) but not the team itself
Assistant coach — same as coach without toggle (the toggle is hidden in UI for assistant role)

Change 3 — Messages redesign (one-way broadcasts)
The current free-form messaging system gets replaced with a focused broadcast system. Coach sends, players read. No replies.
Database
sql-- Drop or repurpose existing message tables if they're truly unused
-- The conversations / messages / conversation_members tables can stay for the existing announcements feature
-- For broadcasts, we'll use the announcements table (already exists)

-- Verify announcements table has what we need
alter table public.announcements
  add column if not exists message_type text default 'announcement'
    check (message_type in ('announcement', 'reminder', 'urgent', 'general')),
  add column if not exists target_audience text default 'all'
    check (target_audience in ('all', 'players_only', 'parents_only', 'specific_players')),
  add column if not exists target_user_ids uuid[],
  add column if not exists channels jsonb default '{"in_app": true, "email": false, "whatsapp": false}'::jsonb,
  add column if not exists email_sent_at timestamptz,
  add column if not exists email_sent_count int default 0,
  add column if not exists whatsapp_opened_at timestamptz,
  add column if not exists whatsapp_opened_count int default 0;

-- Track delivery per recipient (for analytics and read receipts)
create table if not exists public.announcement_deliveries (
  id uuid primary key default uuid_generate_v4(),
  announcement_id uuid not null references announcements(id) on delete cascade,
  user_id uuid not null references auth.users(id) on delete cascade,
  channel text not null check (channel in ('in_app', 'email', 'whatsapp')),
  status text default 'pending' check (status in ('pending', 'sent', 'delivered', 'read', 'failed', 'skipped')),
  sent_at timestamptz,
  read_at timestamptz,
  error_detail text,
  unique(announcement_id, user_id, channel)
);

create index if not exists announcement_deliveries_announcement_idx
  on announcement_deliveries(announcement_id);
create index if not exists announcement_deliveries_user_idx
  on announcement_deliveries(user_id, status);

alter table announcement_deliveries enable row level security;

create policy "users see own deliveries" on announcement_deliveries
  for select to authenticated
  using (user_id = auth.uid());

create policy "coaches see deliveries for their team announcements" on announcement_deliveries
  for select to authenticated
  using (
    exists(
      select 1 from announcements a
      where a.id = announcement_deliveries.announcement_id
        and is_team_coach(a.team_id)
    )
  );

-- User preferences for which channels they receive
alter table public.profiles
  add column if not exists notification_preferences jsonb default '{"in_app": true, "email": true, "whatsapp": true}'::jsonb;

Backend endpoints
GET /api/teams/[teamId]/announcements
  Returns announcements with pagination, optional filter by type/date

POST /api/teams/[teamId]/announcements
  Body: {
    message_type: 'announcement' | 'reminder' | 'urgent' | 'general';
    title: string;
    body: string;
    target_audience: 'all' | 'players_only' | 'parents_only' | 'specific_players';
    target_user_ids?: string[];
    channels: { in_app: boolean; email: boolean; whatsapp: boolean };
    pinned?: boolean;
  }
  Creates announcement, fires deliveries.

GET /api/teams/[teamId]/announcements/[id]/deliveries
  Returns per-recipient delivery status (coach only)

POST /api/teams/[teamId]/announcements/[id]/resend
  Body: { audience: 'unread' | 'all' | 'failed_email' }
  Resends to specified audience subset

POST /api/teams/[teamId]/announcements/[id]/whatsapp-tabs
  Returns array of pre-formatted wa.me URLs for the coach's browser to open

That's the prompt. A few honest notes:
On the can_manage_team_settings toggle being the trickiest part — this is real permission elevation. If the agent gates it only in the UI (hides buttons but doesn't enforce in RLS), you have a security hole where a clever user could call the API directly. The RLS policies I included use a can_manage_team() helper function that does the right thing at the database level. Verify the agent uses this function in RLS, not just in React components.
On the games tab being a rename — the work here is mostly "make the existing event form actually work" plus adding tabs for past/upcoming/next. If the agent says "I need to rebuild the events system," push back — the events table is fine, the form needs all the fields, the page needs all three tabs. Don't let scope creep.
On the messaging design being one-way — this is intentional simplicity. If after a few months you decide you actually want player → coach DMs, that's a future addition. Don't pre-build it. The "Need to reach coach?" footer with WhatsApp/Call shortcuts handles 95% of "I need to ask the coach something" cases without building a chat system.
On WhatsApp delivery being manual — the "open N tabs" approach is the same as before. Coach clicks one button, app opens 15 WhatsApp tabs, coach clicks send in each. Tedious but free. The alternative is paying for WhatsApp Business API (~$25/month minimum + per-message costs). Don't pay until you have paying customers.
On notification preferences being per-player — this is a really important detail that's easy to skip. Some players' parents will hate getting 3 notifications for every message (push + email + WhatsApp). Letting them turn off email or WhatsApp at their preference level keeps the app from feeling spammy. Don't let the agent skip this.
One specific test to do after Change 3 is done — send yourself a broadcast as a coach, including all three channels. Verify:

You see an in-app notification (test on a registered player account, not your admin)
You receive an email in your inbox
The WhatsApp tab opener works (opens correct phone numbers with pre-filled message)

If any of those don't work, the channel logic is broken. Don't ship until all three deliver to a real test player.
