Cause B — Server cookies not cleared
In Next.js App Router with @supabase/ssr, signing out client-side doesn't clear server cookies. Need a server action:
typescript// app/auth/signout/route.ts
import { createServerClient } from '@supabase/ssr';
import { cookies } from 'next/headers';
import { NextResponse } from 'next/server';

export async function POST() {
  const cookieStore = cookies();
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        get(name) { return cookieStore.get(name)?.value; },
        set(name, value, options) { cookieStore.set({ name, value, ...options }); },
        remove(name, options) { cookieStore.set({ name, value: '', ...options }); }
      }
    }
  );

  await supabase.auth.signOut();

  return NextResponse.redirect(new URL('/auth/signin', process.env.NEXT_PUBLIC_APP_URL!), {
    status: 302
  });
}

Bug 3 — Invitation accepts the wrong session (CRITICAL)
This is a serious bug. When admin sends an invitation, then the recipient (a different person, on a different device) clicks the link, they sometimes get logged in as the admin instead of becoming the new user.
If this happens within a single browser (admin clicks own invite link to test), it means session cookies are bleeding into the invite acceptance flow.
Diagnostic

Show me the /invite/[token] route handler
Show me the /api/auth/accept-invite API route
Show me whether the invite page checks for an existing session and what it does

The fix
The invite acceptance flow MUST sign out any existing session before creating the new account. Otherwise:

Server-side rendering of /invite/[token] runs with the existing user's auth context
The invitation token validation might pass (since it's anon-readable), but the page shows the existing user's data
Form submit creates a new user but the session swap fails or partially happens

Server-side sign out on invite landing
typescript// app/invite/[token]/page.tsx
import { createServerClient } from '@supabase/ssr';
import { cookies } from 'next/headers';
import { redirect } from 'next/navigation';

export default async function InvitePage({ params }: { params: { token: string } }) {
  const cookieStore = cookies();

  // Use anon client to fetch invitation (RLS allows this for valid pending invites)
  const supabaseAnon = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    { cookies: { get: (n) => cookieStore.get(n)?.value } }
  );

  const { data: invitation } = await supabaseAnon
    .from('platform_invitations')
    .select('email, invited_role, expires_at, status, notes')
    .eq('token', params.token)
    .single();

  if (!invitation) {
    return <InvalidInvitation reason="not_found" />;
  }
  if (invitation.status !== 'pending') {
    return <InvalidInvitation reason="already_used" />;
  }
  if (new Date(invitation.expires_at) < new Date()) {
    return <InvalidInvitation reason="expired" />;
  }

  // Check current session
  const { data: { user: currentUser } } = await supabaseAnon.auth.getUser();

  if (currentUser) {
    // Someone is already logged in. Two cases:
    if (currentUser.email === invitation.email) {
      // Same user clicking their own invitation — already accepted, redirect to home
      redirect('/');
    } else {
      // Different user — they need to sign out first
      // Show a warning page rather than silently accepting
      return <SignOutFirstScreen
        currentEmail={currentUser.email}
        invitedEmail={invitation.email}
        token={params.token}
      />;
    }
  }

  // No current session — show the accept-invite form
  return <AcceptInviteForm invitation={invitation} token={params.token} />;
}

Server action: accept-invite must use service role and not user's session
typescript// app/api/auth/accept-invite/route.ts
import { createClient } from '@supabase/supabase-js';
import { NextResponse } from 'next/server';

export async function POST(request: Request) {
  const { token, password, full_name } = await request.json();

  // Use SERVICE ROLE — accepting an invite does NOT use the requester's auth
  const supabaseAdmin = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!
  );

  // 1. Validate invitation
  const { data: invitation, error: invErr } = await supabaseAdmin
    .from('platform_invitations')
    .select('*')
    .eq('token', token)
    .eq('status', 'pending')
    .single();

  if (invErr || !invitation) {
    return NextResponse.json({ error: 'invalid_invitation' }, { status: 400 });
  }
  if (new Date(invitation.expires_at) < new Date()) {
    return NextResponse.json({ error: 'expired' }, { status: 400 });
  }

  // 2. Create user via admin API
  const { data: newUser, error: createErr } = await supabaseAdmin.auth.admin.createUser({
    email: invitation.email,
    password,
    email_confirm: true,
    user_metadata: { full_name }
  });

  if (createErr || !newUser?.user) {
    return NextResponse.json({ error: 'user_creation_failed', detail: createErr?.message }, { status: 500 });
  }

  // 3. Update profile (admin role if admin invitation)
  await supabaseAdmin
    .from('profiles')
    .update({
      full_name,
      account_status: 'active',
      is_platform_admin: invitation.invited_role === 'admin',
    })
    .eq('id', newUser.user.id);

  // 4. Mark invitation accepted
  await supabaseAdmin
    .from('platform_invitations')
    .update({
      status: 'accepted',
      accepted_by_user_id: newUser.user.id,
      accepted_at: new Date().toISOString()
    })
    .eq('id', invitation.id);

  // 5. Handle prefill_team_id (multi-coach invites)
  if (invitation.notes) {
    try {
      const notes = typeof invitation.notes === 'string' ? JSON.parse(invitation.notes) : invitation.notes;
      if (notes.prefill_team_id && invitation.invited_role === 'coach') {
        await supabaseAdmin.from('team_members').insert({
          team_id: notes.prefill_team_id,
          user_id: newUser.user.id,
          role: notes.prefill_role || 'coach',
          status: 'active'
        });
      }
    } catch (e) { /* notes wasn't JSON, that's fine */ }
  }

  // 6. Generate a new session for the new user
  // CRITICAL: this must be the new user's session, not the requester's
  const { data: linkData } = await supabaseAdmin.auth.admin.generateLink({
    type: 'magiclink',
    email: invitation.email
  });

  // 7. Return the redirect URL — client uses this to sign in as the new user
  return NextResponse.json({
    success: true,
    redirect_url: '/onboarding/coach',  // or admin/player based on role
    new_user_id: newUser.user.id
  });
}

API
typescript// app/api/admin/invitations/[id]/resend/route.ts
export async function POST(request: Request, { params }: { params: { id: string } }) {
  // Verify admin
  const admin = await requireAdmin();

  // Fetch invitation
  const { data: invitation } = await supabaseAdmin
    .from('platform_invitations')
    .select('*')
    .eq('id', params.id)
    .single();

  if (!invitation) return NextResponse.json({ error: 'not_found' }, { status: 404 });
  if (invitation.status !== 'pending') {
    return NextResponse.json({ error: 'not_pending' }, { status: 400 });
  }
  if (invitation.email_send_count >= 5) {
    return NextResponse.json({ error: 'max_resends_reached' }, { status: 429 });
  }

  // Send email
  const inviteUrl = `${process.env.NEXT_PUBLIC_APP_URL}/invite/${invitation.token}`;
  const result = await sendInvitationEmail({
    to: invitation.email,
    inviteUrl,
    inviterName: admin.full_name,
    language: 'he'
  });

  if (result.error) {
    return NextResponse.json({
      error: 'email_failed',
      detail: result.error.message
    }, { status: 500 });
  }

  // Update send count
  await supabaseAdmin
    .from('platform_invitations')
    .update({
      email_sent_at: new Date().toISOString(),
      email_send_count: invitation.email_send_count + 1
    })
    .eq('id', invitation.id);

  // Audit log
  await supabaseAdmin.from('admin_audit_log').insert({
    admin_id: admin.id,
    action: 'invitation_resent',
    metadata: { invitation_id: invitation.id, email: invitation.email, attempt: invitation.email_send_count + 1 }
  });

  return NextResponse.json({ success: true });
}
