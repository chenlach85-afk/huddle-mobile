-- Update the handle_new_user trigger to gate signups
create or replace function public.handle_new_user()
returns trigger
language plpgsql
security definer set search_path = public
as $$
declare
  has_platform_invite boolean;
  has_team_code text;
begin
  -- Check for valid platform invitation
  select exists(
    select 1 from platform_invitations
    where email = new.email and status = 'pending' and expires_at > now()
  ) into has_platform_invite;

  -- Check for team join via metadata (set during signup-via-team-code flow)
  has_team_code := new.raw_user_meta_data->>'joining_team_id';

  -- Allow if: bootstrap (no profiles yet) OR platform invite OR team code in metadata
  if not has_platform_invite
     and has_team_code is null
     and (select count(*) from profiles) > 0 then
    raise exception 'Signup requires either a platform invitation or a valid team invite code';
  end if;

  insert into public.profiles (id, email, full_name, account_status)
  values (
    new.id,
    new.email,
    coalesce(new.raw_user_meta_data->>'full_name', ''),
    'active'  -- auto-approved
  );
  return new;
end;
$$;

drop trigger if exists on_auth_user_created on auth.users;
