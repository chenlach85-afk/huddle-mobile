async function addPlayer(req) {
  // 1. Verify caller is coach of this team
  // 2. Validate: at least full_name; if invitation_action != 'none', email or phone required
  // 3. Insert team_members placeholder row:
  const { data: member } = await supabase.from('team_members').insert({
    team_id: teamId,
    user_id: null,
    role: 'player',
    placeholder_full_name: full_name,
    placeholder_email: email,
    placeholder_phone: phone,
    jersey_number,
    position,
    notes,
    status: invitation_action === 'none' ? 'pending_invitation' : 'invited'
  }).select().single();

  // 4. If invitation requested, create the invitation:
  if (invitation_action !== 'none') {
    if (!email && !phone) throw new Error('Email or phone required for invitation');

    const { data: invitation } = await supabase.from('team_invitations').insert({
      team_id: teamId,
      email,
      phone,
      invited_role: 'player',
      prefill_full_name: full_name,
      prefill_jersey_number: jersey_number,
      prefill_position: position,
      personal_message,
      invited_by: caller.id,
      status: 'pending'
    }).select().single();

    // Link the team_member row to the invitation
    await supabase.from('team_members').update({
      invitation_id: invitation.id
    }).eq('id', member.id);

    // Send the email if requested
    if (invitation_action === 'send_email') {
      await sendTeamInvitationEmail({ to: email, invitation, team, coach });
      await supabase.from('team_invitations').update({
        email_sent_at: new Date(),
        email_send_count: 1
      }).eq('id', invitation.id);
    }
    // For 'generate_link', just return the link in the response
  }

  return { player: member, invitation, invite_url: invitation ? `${APP_URL}/team-invite/${invitation.token}` : null };

import { NextResponse } from 'next/server';

export async function GET() {
  const apiKey = process.env.RESEND_API_KEY;
  const fromEmail = process.env.RESEND_FROM_EMAIL;

  const diagnostic: any = {
    env: {
      RESEND_API_KEY_exists: !!apiKey,
      RESEND_API_KEY_length: apiKey?.length,
      RESEND_API_KEY_prefix: apiKey?.substring(0, 4),
      RESEND_FROM_EMAIL: fromEmail,
      NODE_ENV: process.env.NODE_ENV,
    }
  };

  if (!apiKey) {
    return NextResponse.json({ ...diagnostic, error: 'NO_API_KEY' });
  }

  // Test 1: direct fetch to Resend API
  try {
    const directResponse = await fetch('https://api.resend.com/emails', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        from: fromEmail || 'HUDDLE <onboarding@resend.dev>',
        to: ['delivered@resend.dev'], // Resend's test inbox that always accepts
        subject: 'HUDDLE diagnostic test',
        html: '<p>Test from diagnostic endpoint</p>'
      })
    });

    diagnostic.direct_fetch = {
      status: directResponse.status,
      statusText: directResponse.statusText,
      body: await directResponse.json().catch(() => null),
    };
  } catch (e: any) {
    diagnostic.direct_fetch = { error: e.message, name: e.name };
  }

  // Test 2: via Resend SDK
  try {
    const resend = new Resend(apiKey);
    const sdkResult = await resend.emails.send({
      from: fromEmail || 'HUDDLE <onboarding@resend.dev>',
      to: ['delivered@resend.dev'],
      subject: 'HUDDLE diagnostic test (SDK)',
      html: '<p>Test from SDK</p>'
    });
    diagnostic.sdk_send = sdkResult;
  } catch (e: any) {
    diagnostic.sdk_send = { error: e.message, name: e.name, stack: e.stack };
  }

  return NextResponse.json(diagnostic);
