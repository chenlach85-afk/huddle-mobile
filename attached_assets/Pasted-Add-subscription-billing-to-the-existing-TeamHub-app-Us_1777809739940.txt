-- ============================================================================
-- SUBSCRIPTIONS — one per paying coach
-- ============================================================================
create table public.subscriptions (
  id uuid primary key default uuid_generate_v4(),
  user_id uuid not null references auth.users(id) on delete cascade,
  plan_id text not null references plans(id),
  billing_period text default 'monthly' check (billing_period in ('monthly', 'yearly')),
  stripe_customer_id text,
  stripe_subscription_id text unique,
  status text not null check (status in ('trialing', 'active', 'past_due', 'canceled', 'incomplete', 'incomplete_expired', 'unpaid', 'paused')),
  trial_ends_at timestamptz,
  current_period_start timestamptz,
  current_period_end timestamptz,
  cancel_at_period_end boolean default false,
  canceled_at timestamptz,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique(user_id)  -- one active sub per user
);

create index subscriptions_status_idx on subscriptions(status);
create index subscriptions_stripe_customer_idx on subscriptions(stripe_customer_id);

-- ============================================================================
-- COACH_INVITATIONS — admin-issued invites for comp accounts or beta testers
-- ============================================================================
create table public.coach_invitations (
  id uuid primary key default uuid_generate_v4(),
  email text not null,
  invited_by uuid references auth.users(id),

  trial_days int default 14,
  comp boolean default false,         -- if true, free forever

  used_by uuid references auth.users(id),
  used_at timestamptz,
  expires_at timestamptz default now() + interval '30 days',
  notes text,
  created_at timestamptz default now()
);

create index coach_invitations_email_idx on coach_invitations(email);
create index coach_invitations_token_idx on coach_invitations(token);

-- ============================================================================
-- HELPER FUNCTION: check if user has active subscription
-- ============================================================================
create or replace function public.has_active_subscription(check_user_id uuid)
returns boolean language sql security definer stable as $$
  select exists(
    select 1 from subscriptions
    where user_id = check_user_id
      and status in ('trialing', 'active')
      and (current_period_end is null or current_period_end > now())
  );
$$;

create or replace function public.get_user_plan(check_user_id uuid)
returns text language sql security definer stable as $$
  select plan_id from subscriptions
  where user_id = check_user_id
    and status in ('trialing', 'active')
  limit 1;
$$;

-- ============================================================================
-- AUTO-CREATE TRIAL SUBSCRIPTION when user creates their first team
-- ============================================================================
create or replace function public.start_trial_on_team_create()
returns trigger language plpgsql security definer set search_path = public as $$
begin
  -- Only create trial if user doesn't already have a subscription
  if not exists(select 1 from subscriptions where user_id = new.created_by) then
    insert into subscriptions (user_id, plan_id, status, trial_ends_at)
    values (new.created_by, 'solo', 'trialing', now() + interval '14 days');
  end if;
  return new;
end; $$;

drop trigger if exists on_team_created_start_trial on teams;
create trigger on_team_created_start_trial
  after insert on teams
  for each row execute procedure public.start_trial_on_team_create();

Critical rules

Existing MVP must keep working unchanged. All current features (events, messages, tasks, etc.) work for free during the trial.
No paywalls during trial. Coaches see all features.
Hard limits only after trial expires. Soft warnings (banners) before.
Player experience never changes. Players are always free; they don't see billing UI at all.
Multi-coach support per team is gated by plan (max_coaches_per_team).
Stripe is source of truth for status. Webhook is the only thing that flips status to active or canceled. Never trust the client.
Comp accounts: subscriptions.stripe_subscription_id IS NULL AND status = 'active' means it's a comp — don't try to fetch from Stripe.
Test mode vs live mode: Use STRIPE_SECRET_KEY value's prefix (sk_test_ vs sk_live_) to detect mode. Show "Test Mode" banner in admin if test.

What NOT to do

Don't add billing to the player experience
Don't break free trial (a hard ask for cards on day 1 will kill conversion)
Don't put Stripe.js on every page (only checkout and billing pages)
Don't store credit card numbers (Stripe handles this — never touch raw card data)
Don't auto-charge after trial without warning emails on day 11
Don't lock data when subscription expires (preserve everything; just disable write actions)
