Fix B: Subscription flow not running
Verify the client subscribes on app load. Common implementation:
typescript// hooks/usePushSubscription.ts
'use client';

import { useEffect } from 'react';
import { createBrowserClient } from '@supabase/ssr';

export function usePushSubscription() {
  useEffect(() => {
    if (!('serviceWorker' in navigator) || !('PushManager' in window)) return;

    async function subscribe() {
      const reg = await navigator.serviceWorker.ready;
      let sub = await reg.pushManager.getSubscription();

      if (!sub) {
        // Ask permission first
        const permission = await Notification.requestPermission();
        if (permission !== 'granted') return;

        sub = await reg.pushManager.subscribe({
          userVisibleOnly: true,
          applicationServerKey: urlBase64ToUint8Array(
            process.env.NEXT_PUBLIC_VAPID_PUBLIC_KEY!
          )
        });
      }

      // Save subscription to profile
      const supabase = createBrowserClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
      );
      await supabase
        .from('profiles')
        .update({ push_subscription: sub.toJSON() })
        .eq('id', (await supabase.auth.getUser()).data.user?.id);
    }

    subscribe().catch(console.error);
  }, []);
}

function urlBase64ToUint8Array(base64: string) {
  const padding = '='.repeat((4 - (base64.length % 4)) % 4);
  const data = (base64 + padding).replace(/-/g, '+').replace(/_/g, '/');
  const raw = atob(data);
  return new Uint8Array([...raw].map(c => c.charCodeAt(0)));
}

typescriptimport webpush from 'web-push';
import { supabaseAdmin } from './supabase-admin';

webpush.setVapidDetails(
  'mailto:admin@huddle.app',
  process.env.NEXT_PUBLIC_VAPID_PUBLIC_KEY!,
  process.env.VAPID_PRIVATE_KEY!
);

export async function sendPushToUsers(userIds: string[], payload: {
  title: string;
  body: string;
  url?: string;
}) {
  const { data: profiles } = await supabaseAdmin
    .from('profiles')
    .select('id, push_subscription')
    .in('id', userIds)
    .not('push_subscription', 'is', null);

  if (!profiles?.length) return { sent: 0, failed: 0 };

  let sent = 0, failed = 0;
  await Promise.all(profiles.map(async (p) => {
    try {
      await webpush.sendNotification(p.push_subscription, JSON.stringify(payload));
      sent++;
    } catch (err: any) {
      failed++;
      // Subscription expired/revoked
      if (err.statusCode === 410) {
        await supabaseAdmin
          .from('profiles')
          .update({ push_subscription: null })
          .eq('id', p.id);
      }
      console.error(`Push send failed for ${p.id}:`, err.statusCode, err.body);
    }
  }));

  return { sent, failed };
}

Israeli phone number normalization
Israeli numbers come in multiple formats:

050-1234567 (with hyphens)
0501234567 (no hyphens)
+972501234567 (international)
972-50-1234567 (international with hyphens)
972501234567 (international, no formatting)

Normalize to international without + for wa.me URLs:
typescriptexport function normalizePhoneForWhatsApp(phone: string): string | null {
  const digits = phone.replace(/\D/g, '');

  if (digits.startsWith('972')) {
    return digits;
  }

  if (digits.startsWith('0') && digits.length === 10) {
    // Israeli local: 0501234567 → 972501234567
    return '972' + digits.substring(1);
  }

  return null; // Invalid
}
Validate phone numbers on input. Show user-friendly error: "Phone number doesn't look like a valid Israeli mobile number."

That's the full prompt. Some honest practical notes:
On notifications first — I put Change 4 first because everything else builds on it. If notifications don't work, none of the cool features (event broadcast, team invitations, message alerts) feel right. Fix the foundation before piling on.
On WhatsApp's manual click limitation — to be clear, the "open N tabs" approach isn't elegant. It's the free, no-API-key way to do it. The coach has to click "send" in each tab. For a 15-player team, that's 15 clicks. Tedious but free. The alternative is paying for WhatsApp Business API (~$25/month minimum), which is overkill until you have paying customers.
On the test push button — actually click this when the agent says "done." If a notification doesn't appear on your phone within 5 seconds, the agent didn't really finish. Don't ship to coaches until you can reliably get a test notification through.
On event types specifically — once these are color-coded, your calendar will look much more professional immediately. This is one of those changes where the visual improvement is way out of proportion to the code change. Worth doing well.
